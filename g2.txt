        // Generate captcha
        function generateCaptcha() {
            const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789';
            currentCaptcha = '';
            for (let i = 0; i < 6; i++) {

        // Handle login
        function handleLogin(e) {
            e.preventDefault();
            
            const countryCode = document.getElementById('loginCountryCode').value;
            const phone = document.getElementById('loginPhone').value;
            const fullPhone = countryCode + phone;
            const password = document.getElementById('loginPassword').value;
            
            // Check if user exists in localStorage
            const users = JSON.parse(localStorage.getItem('registeredUsers') || '[]');
            const user = users.find(u => u.phone === fullPhone && u.password === password);
            
            if (user) {
                if (user.phoneVerified) {
                    currentUser = user;
                    // Ensure claims array exists for the user
                    if (!currentUser.claims) {
                        currentUser.claims = [];
                    }
                    localStorage.setItem('currentUser', JSON.stringify(currentUser));
                    hideLoginModal();
                    updateUIForLoggedInUser();
                } else {
                    alert('Please verify your phone number first.');
                    showOtpModal(fullPhone);
                }
            } else {
                alert('Invalid phone number or password. Please try again.');
            }
        }

        // Handle signup
        function handleSignup(e) {
            e.preventDefault();
            
            const name = document.getElementById('signupName').value;
            const email = document.getElementById('signupEmail').value;
            const countryCode = document.getElementById('signupCountryCode').value;
            const phone = document.getElementById('signupPhone').value;
            const fullPhone = countryCode + phone;
            const country = document.getElementById('signupCountry').value;
            const address = document.getElementById('signupAddress').value;
            const password = document.getElementById('signupPassword').value;
            const captchaInput = document.getElementById('captchaInput').value;
            
            // Validate captcha
            if (captchaInput !== currentCaptcha) {
                alert('Invalid captcha. Please try again.');
                generateCaptcha();
                return;
            }
            
            // Check if user already exists
            const users = JSON.parse(localStorage.getItem('registeredUsers') || '[]');
            if (users.find(u => u.phone === fullPhone || u.email === email)) {
                alert('User with this phone number or email already exists.');
                return;
            }
            
            // Create new user
            const newUser = {
                name,
                email,
                phone: fullPhone,
                country,
                address,
                password,
                phoneVerified: false,
                claims: [], // Start with an empty claims array
                registeredAt: new Date().toISOString()
            };
            
            users.push(newUser);
            localStorage.setItem('registeredUsers', JSON.stringify(users));
            
            // Proceed to OTP verification
            currentUser = newUser;
            showOtpModal(fullPhone);
        }

        // Handle OTP verification
        function handleOtpVerification(e) {
            e.preventDefault();
            
            const otpInputs = document.querySelectorAll('.otp-input');
            const enteredOtp = Array.from(otpInputs).map(input => input.value).join('');
            
            if (enteredOtp === generatedOtp) {
                // Mark phone as verified
                if (currentUser) {
                    currentUser.phoneVerified = true;
                    
                    // Update in localStorage
                    const users = JSON.parse(localStorage.getItem('registeredUsers') || '[]');
                    const userIndex = users.findIndex(u => u.phone === currentUser.phone);
                    if (userIndex !== -1) {
                        users[userIndex] = currentUser;
                        localStorage.setItem('registeredUsers', JSON.stringify(users));
                    }
                    
                    localStorage.setItem('currentUser', JSON.stringify(currentUser));
                    
                    // Clear timer
                    if (otpTimer) {
                        clearInterval(otpTimer);
                    }
                    
                    hideOtpModal();
                    updateUIForLoggedInUser();
                }
            } else {
                alert('Invalid OTP. Please try again.');
                // Clear OTP inputs
                otpInputs.forEach(input => input.value = '');
                otpInputs[0].focus();
            }
        }

        // Start OTP timer
        function startOtpTimer() {
            let timeLeft = 300; // 5 minutes
            const timerDisplay = document.getElementById('otpTimer');
            
            if (otpTimer) {
                clearInterval(otpTimer);
            }
            
            otpTimer = setInterval(() => {
                const minutes = Math.floor(timeLeft / 60);
                const seconds = timeLeft % 60;
                timerDisplay.textContent = `${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`;
                
                if (timeLeft <= 0) {
                    clearInterval(otpTimer);
                    alert('OTP expired. Please request a new one.');
                }
                
                timeLeft--;
            }, 1000);
        }

                    supportDropZone.classList.remove('border-blue-400', 'bg-blue-50');
                });
                
                supportDropZone.addEventListener('drop', (e) => {
                    e.preventDefault();
                    supportDropZone.classList.remove('border-blue-400', 'bg-blue-50');
                    const files = e.dataTransfer.files;
                    handleSupportFiles(files);
                });
                
                supportFileInput.addEventListener('change', (e) => {
                    handleSupportFiles(e.target.files);
                });
            }
        }
        
        // Handle support file uploads
        function handleSupportFiles(files) {
            const fileList = Array.from(files);
            if (fileList.length > 0) {
                const fileNames = fileList.map(file => file.name).join(', ');
                const dropZone = document.querySelector('#supportPage input[type="file"]')?.parentElement;
                if (dropZone) {
                    const textElement = dropZone.querySelector('p');
                    if (textElement) {
                        textElement.textContent = `Selected: ${fileNames}`;
                    }
                }
            }
        }
        
        // Show support success message
        function showSupportSuccessMessage(ticketNumber, priority) {
            const priorityColors = {
                'low': 'from-green-500 to-green-600',
                'medium': 'from-yellow-500 to-yellow-600',
                'high': 'from-red-500 to-red-600'
            };
            
            const priorityText = {
                'low': '24-48 hours',
                'medium': '12-24 hours',
                'high': '2-4 hours'
            };
            
            const successHtml = `
                <div id="supportSuccessMessage" class="fixed top-4 right-4 bg-gradient-to-r ${priorityColors[priority]} text-white px-6 py-4 rounded-lg shadow-lg z-50 max-w-sm">
                    <div class="flex items-start">
                        <div class="flex-shrink-0">
                            <svg class="w-6 h-6 mt-0.5" fill="currentColor" viewBox="0 0 24 24">
                                <path d="M9 16.17L4.83 12l-1.42 1.41L9 19 21 7l-1.41-1.41z"/>
                            </svg>
                        </div>
                        <div class="ml-3 flex-1">
                            <h4 class="font-semibold">Message Sent Successfully!</h4>
                            <p class="text-sm opacity-90 mt-1">Ticket #${ticketNumber}</p>
                            <p class="text-xs opacity-80 mt-1">Expected response: ${priorityText[priority]}</p>
                        </div>
                        <button onclick="closeSupportSuccessMessage()" class="ml-2 opacity-70 hover:opacity-100">
                            <svg class="w-5 h-5" fill="currentColor" viewBox="0 0 24 24">
                                <path d="M19 6.41L17.59 5 12 10.59 6.41 5 5 6.41 10.59 12 5 17.59 6.41 19 12 13.41 17.59 19 19 17.59 13.41 12z"/>
                            </svg>
                        </button>
                    </div>
                </div>
            `;
            
            // Remove any existing success message
            const existingMessage = document.getElementById('supportSuccessMessage');
            if (existingMessage) {
                existingMessage.remove();
            }
            
            // Add new success message
            document.body.insertAdjacentHTML('beforeend', successHtml);
            
            // Auto-hide after 8 seconds
            setTimeout(() => {
                closeSupportSuccessMessage();
            }, 8000);
        }
        
        function closeSupportSuccessMessage() {
            const successMessage = document.getElementById('supportSuccessMessage');
            if (successMessage) {
                successMessage.style.transform = 'translateX(100%)';
                successMessage.style.transition = 'transform 0.3s ease-out';
                setTimeout(() => {
                    successMessage.remove();
                }, 300);
            }
        }
    </script>
